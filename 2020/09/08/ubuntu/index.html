ubuntu

apt命令
sudo apt-get update # 更新本地包数据库
sudo apt-get upgrade # 更新所有已安装的包
sudo apt-get install # 安装
sudo apt-get autoremove # 自动移除不需要的包
补充几个常用的清理命令：
apt-get autoclean: 将已删除软件包的.deb安装文件从硬盘中删除；
apt-get clean: 同上，但会把已安装的软件包的安装包也删除掉；
apt-get autoremove: 删除为了满足其他软件包的依赖而安装，但现在不再需要的软件包；
apt-get remove : 删除已安装的软件包（保留配置文件）；
apt-get –purge remove [软件包名]: 删除已安装包（不保留配置文件）
sudo apt-cache search package_name # 搜索软件
sudo apt-cache show package_name # 查看软件包信息
sudo apt-cache show depends package_name # 查看软件包依赖关系

安装 vsftpd
sudo apt-get install vsftpd
vi /etc/vsftpd.conf

systemctl status vsftpd

安装 Apache
sudo apt update
sudo apt install apache2
sudo ufw allow ‘Apache Full’ #防火墙中打开80,443端口
sudo ufw status #查看更改
vi /etc/apache2/sites-eabled/000-default.cong #配置文件
sudo systemctl restart appache2
sudo sustemctl status appache2
管理apache服务
sudo /etc/init.d/apache2 [ start | stop | restart | status ]
或
service apache2 [ start | stop | restart | status ]

防火墙
iptables -F #关闭
sudo ufw enable|disable #开启/关闭防火墙
sudo ufw default deny #开启了防火墙，并在系统启动时自动开启，关闭所有外部对本机的访问，但本机访问外部正常。
sudo ufw allow/deny 20 #开启/禁用20端口
sudo ufw default allow/deny #外来访问默认允许/拒绝
sudo ufw delete allow 53 #删除53端口的规则
sudo ufw status #查看
